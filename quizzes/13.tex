\begin{schemedisplay}
  (define-interface lunch<%> [lunch?])

  (define (yes-mixin %)
    (class
      (super % lunch<%>)
      (define/public (lunch?)
        "Yes!")))

  (define (no-mixin %)
    (class
      (super % lunch<%>)
      (define/public (lunch?)
        "No, thanks.")))

  (define (dunno-mixin %)
    (class
      (super % lunch<%>)
      (define/public (lunch?)
        (string-append "Uhh... " (super)))))

  (define dude
    (new (dunno-mixin
           (yes-mixin
             (dunno-mixin
               (no-mixin
                 (class
                   (define/public (lunch?)
                     "What?"))))))))
  (dude . lunch?)
\end{schemedisplay}

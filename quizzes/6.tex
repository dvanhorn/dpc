What is the result of this program?

\bigskip

\begin{schemedisplay}
  ; A Box is a (box\% Number)
  (define-class box%
    (fields data)

    (define/public (grow)
      (begin
        (set-field! data (+ 1 (field data)))
        (field data))))

  (define a (box% 0))
  (+ (a . grow) (a . grow) (a . grow))
\end{schemedisplay}
